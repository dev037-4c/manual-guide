<script>
(() => {
  const root = document.currentScript.previousElementSibling.previousElementSibling;
  const track = root.querySelector('.track');
  const prev = root.querySelector('.prev');
  const next = root.querySelector('.next');
  const dotsWrap = root.querySelector('.dots');
  const slides = Array.from(track.children);
  const step = root.offsetWidth > 480 ? 336 : root.offsetWidth*0.78;
  let index = 0;
  let autoplayId = null;
  const autoplay = root.dataset.autoplay === 'true';
  const interval = parseInt(root.dataset.interval||'4000',10);

  slides.forEach((_,i)=>{
    const b=document.createElement('button');
    b.addEventListener('click',()=>goTo(i));
    dotsWrap.appendChild(b);
  });

  function update(){
    track.style.transform=`translateX(${-index*step}px)`;
    dotsWrap.querySelectorAll('button').forEach((b,i)=>b.setAttribute('aria-current',i===index));
  }
  function goTo(i){ index=(i+slides.length)%slides.length; update(); }
  function nextFn(){ goTo(index+1); }
  function prevFn(){ goTo(index-1); }
  prev.addEventListener('click',prevFn);
  next.addEventListener('click',nextFn);

  function start(){ if(!autoplay)return; stop(); autoplayId=setInterval(nextFn,interval);}
  function stop(){ if(autoplayId) clearInterval(autoplayId); }
  root.addEventListener('mouseenter',stop);
  root.addEventListener('mouseleave',start);

  goTo(0); start();
})();
</script>
