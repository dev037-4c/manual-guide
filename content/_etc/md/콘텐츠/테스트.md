<!-- 카드 슬라이더: JS -->
<script>
(() => {
  const slider = document.currentScript.previousElementSibling.previousElementSibling.previousElementSibling; // <div class="x-slider">
  // 안전장치: 예상 못 찾으면 클래스 기반 탐색
  const root = slider?.classList?.contains('x-slider') ? slider : document.querySelector('.x-slider');
  if (!root) return;

  const viewport = root.querySelector('.x-viewport');
  const track = root.querySelector('.x-track');
  const prevBtn = root.querySelector('.x-prev');
  const nextBtn = root.querySelector('.x-next');
  const dotsWrap = root.querySelector('.x-dots');
  const cards = Array.from(track.children);

  // 한 화면에 보이는 카드 수 근사값(간격 고려)
  const gap = parseInt(getComputedStyle(track).gap) || 16;
  const cardW = parseInt(getComputedStyle(root).getPropertyValue('--card-w')) || 260;
  const step = cardW + gap;

  let index = 0;
  const maxIndex = Math.max(0, cards.length - 1);
  let autoplayId = null;
  const autoplay = root.dataset.autoplay === 'true';
  const interval = parseInt(root.dataset.interval || '4000', 10);

  // 도트 생성
  cards.forEach((_, i) => {
    const b = document.createElement('button');
    b.type = 'button';
    b.setAttribute('aria-label', `${i + 1}번 슬라이드`);
    b.addEventListener('click', () => goTo(i));
    dotsWrap.appendChild(b);
  });

  function update() {
    track.style.transform = `translateX(${-index * step}px)`;
    dotsWrap.querySelectorAll('button').forEach((d, i) => {
      d.setAttribute('aria-current', i === index ? 'true' : 'false');
    });
  }

  function goTo(i) {
    index = Math.max(0, Math.min(i, maxIndex));
    update();
  }

  function next() { goTo(index + 1 > maxIndex ? 0 : index + 1); }
  function prev() { goTo(index - 1 < 0 ? maxIndex : index - 1); }

  nextBtn.addEventListener('click', next);
  prevBtn.addEventListener('click', prev);

  // 키보드 탐색
  root.tabIndex = 0;
  root.addEventListener('keydown', (e) => {
    if (e.key === 'ArrowRight') next();
    if (e.key === 'ArrowLeft') prev();
  });

  // 터치 스와이프
  let sx = 0, dx = 0;
  viewport.addEventListener('touchstart', (e) => { sx = e.touches[0].clientX; dx = 0; }, { passive: true });
  viewport.addEventListener('touchmove',  (e) => { dx = e.touches[0].clientX - sx; }, { passive: true });
  viewport.addEventListener('touchend',   () => {
    if (Math.abs(dx) > 40) (dx < 0 ? next : prev)();
    sx = dx = 0;
  });

  // 자동재생
  function start() {
    if (!autoplay) return;
    stop();
    autoplayId = setInterval(next, interval);
  }
  function stop() {
    if (autoplayId) { clearInterval(autoplayId); autoplayId = null; }
  }
  root.addEventListener('mouseenter', stop);
  root.addEventListener('mouseleave', start);

  // 초기 렌더
  goTo(0);
  start();
})();
</script>
