<script>
(() => {
  const root = document.currentScript.previousElementSibling.previousElementSibling;
  const viewport = root.querySelector('.viewport');
  const track = root.querySelector('.track');
  const prev = root.querySelector('.prev');
  const next = root.querySelector('.next');
  const dotsWrap = root.querySelector('.dots');
  const slides = Array.from(track.children);

  let index = 0;
  let autoplayId = null;
  const autoplay = root.dataset.autoplay === 'true';
  const interval = parseInt(root.dataset.interval || '4000', 10);

  // 도트 생성
  slides.forEach((_, i) => {
    const b = document.createElement('button');
    b.type = 'button';
    b.setAttribute('aria-label', `${i+1}번`);
    b.addEventListener('click', () => goTo(i));
    dotsWrap.appendChild(b);
  });

  // 현재 뷰포트 폭 기준으로 정확히 한 장씩 이동
  function stepWidth() { return viewport.clientWidth; }

  function update() {
    track.style.transform = `translateX(${-index * stepWidth()}px)`;
    dotsWrap.querySelectorAll('button').forEach((d, i) => {
      d.setAttribute('aria-current', i === index ? 'true' : 'false');
    });
  }

  function goTo(i) {
    index = (i + slides.length) % slides.length;
    update();
  }
  function nextFn() { goTo(index + 1); }
  function prevFn() { goTo(index - 1); }

  next.addEventListener('click', nextFn);
  prev.addEventListener('click', prevFn);

  // 리사이즈 시 위치 보정 (반응형)
  window.addEventListener('resize', update);

  // 스와이프
  let sx = 0, dx = 0;
  viewport.addEventListener('touchstart', e => { sx = e.touches[0].clientX; dx = 0; }, { passive: true });
  viewport.addEventListener('touchmove', e => { dx = e.touches[0].clientX - sx; }, { passive: true });
  viewport.addEventListener('touchend', () => {
    if (Math.abs(dx) > 40) (dx < 0 ? nextFn : prevFn)();
    sx = dx = 0;
  }, { passive: true });

  // 자동재생
  function start() { if (!autoplay) return; stop(); autoplayId = setInterval(nextFn, interval); }
  function stop() { if (autoplayId) { clearInterval(autoplayId); autoplayId = null; } }
  root.addEventListener('mouseenter', stop);
  root.addEventListener('mouseleave', start);

  // 초기화
  goTo(0);
  start();
})();
</script>
